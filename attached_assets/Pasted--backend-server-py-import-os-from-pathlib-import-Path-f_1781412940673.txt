# backend/server.py

import os
from pathlib import Path

from fastapi import FastAPI, APIRouter
from fastapi.staticfiles import StaticFiles
from fastapi.responses import FileResponse, JSONResponse
import httpx

# --- Config ---

EODHD_API_KEY = os.getenv("EODHD_API_KEY")

# Map your dashboard "markets" to EODHD symbols
INDEX_MAP = {
    # USA
    "nasdaq100": "NDX.INDX",
    "sp500": "GSPC.INDX",
    "dowjones": "DJI.INDX",

    # Canada
    "tsxcomp": "GSPTSE.INDX",

    # Brazil
    "bovespa": "BVSP.INDX",

    # Europe
    "ftse100": "FTSE.INDX",
    "dax40": "GDAXI.INDX",
    "cac40": "FCHI.INDX",
    "eurostoxx50": "STOXX50E.INDX",
    "smi": "SSMI.INDX",
    "aex": "AEX.INDX",
    "ibex35": "IBEX.INDX",

    # Asia-Pacific
    "nikkei225": "N225.INDX",
    "asx200": "AXJO.INDX",
    "hangseng": "HSI.INDX",
    "shanghaicomp": "SSEC.INDX",
    "sensex": "BSESN.INDX",
    "kospi": "KS11.INDX",

    # Middle East & Africa
    "tadawul": "TASI.INDX",
    "jsetop40": "JTOPI.INDX",
}

# --- FastAPI app ---

app = FastAPI()
router = APIRouter()


# Health check
@router.get("/api/health")
async def health():
    return {"status": "ok"}


# Live index endpoint for dashboard
@router.get("/api/live/{market}")
async def get_live_market(market: str):
    if not EODHD_API_KEY:
        return JSONResponse(
            status_code=500,
            content={"error": "EODHD_API_KEY not configured"},
        )

    symbol = INDEX_MAP.get(market.lower())
    if not symbol:
        return JSONResponse(
            status_code=404,
            content={"error": f"Unknown market '{market}'"},
        )

    url = f"https://eodhd.com/api/real-time/{symbol}"
    params = {
        "api_token": EODHD_API_KEY,
        "fmt": "json",
    }

    try:
        async with httpx.AsyncClient(timeout=5.0) as client:
            r = await client.get(url, params=params)
            r.raise_for_status()
            data = r.json()
    except Exception as e:
        return JSONResponse(
            status_code=502,
            content={"error": "Failed to fetch from EODHD", "detail": str(e)},
        )

    # EODHD real-time response typically includes:
    # close, change, change_p, timestamp, etc.
    price = data.get("close")
    change = data.get("change")
    change_p = data.get("change_p")
    timestamp = data.get("timestamp")

    # ALN = alignment / direction arrow for your UI
    if change is None:
        aln = "→"
    elif change > 0:
        aln = "↑"
    elif change < 0:
        aln = "↓"
    else:
        aln = "→"

    return {
        "market": market,
        "symbol": symbol,
        "price": price,
        "change": change,
        "change_p": change_p,
        "timestamp": timestamp,
        "aln": aln,
    }


# Attach router
app.include_router(router)


# --- Serve frontend (static dashboard) ---

# If you have a build: frontend/build
# If you only have public: frontend/public
FRONTEND_DIR = Path(__file__).parent.parent / "frontend" / "public"

if not FRONTEND_DIR.exists():
    # fallback to build if you build with craco
    FRONTEND_DIR = Path(__file__).parent.parent / "frontend" / "build"

app.mount(
    "/",
    StaticFiles(directory=str(FRONTEND_DIR), html=True),
    name="frontend",
)


@app.get("/")
async def index():
    index_path = FRONTEND_DIR / "index.html"
    return FileResponse(index_path)
